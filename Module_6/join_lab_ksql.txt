 1. In KSQL
 
CREATE STREAM LeftStreamTopic2 (id INT KEY, left_str VARCHAR, left_ts VARCHAR) WITH (KAFKA_TOPIC='LeftStreamTopic2', VALUE_FORMAT='JSON', TIMESTAMP='left_ts', TIMESTAMP_FORMAT='yyyy-MM-dd''T''HH:mm:ssX', PARTITIONS=4);
 
CREATE STREAM RightStreamTopic2 (ID INT KEY, right_str VARCHAR, right_ts VARCHAR) WITH (KAFKA_TOPIC='RightStreamTopic2', VALUE_FORMAT='JSON', TIMESTAMP='right_ts', TIMESTAMP_FORMAT='yyyy-MM-dd''T''HH:mm:ssX', PARTITIONS=4);
